# UI Rework Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the dark NEURAL GRID theme with a soft pastel visual system and implement 10 new features including the Wordle-style board, flip animation, color elimination tracker, and Easy mode.

**Architecture:** All visual constants move to a `PASTEL` constant dictionary in Main.gd. The Wordle board replaces the current history-scroll + active-row architecture by building all rows at once and tracking which row is "active". New features (tracker, hint rework, dot burst, color-blind mode, share, custom puzzle, weekly leaderboard sim) are added as isolated methods in Main.gd.

**Tech Stack:** Godot 4.6, GDScript, StyleBoxFlat for panel styling, Tween for animations, GPUParticles2D for dot burst, Line2D for dot trail.

**Spec:** `docs/superpowers/specs/2026-04-19-ui-rework-design.md`

**Dependency order:** Complete tasks in order — visual foundation first, then board architecture, then features.

```gdscript
func _build_board_row(row_index: int, num_slots: int) -> Dictionary:
	var hbox := HBoxContainer.new()
	hbox.name = "BoardRow%d" % row_index
	hbox.add_theme_constant_override("separation", 6)
	hbox.alignment = BoxContainer.ALIGNMENT_CENTER

	var slots: Array = []
	for s in range(num_slots):
		var slot := SlotButton.new() if ClassDB.class_exists("SlotButton") else Button.new()
		slot.custom_minimum_size = Vector2(56, 56)
		hbox.add_child(slot)
		slots.append(slot)

	# Pip container (right side)
	var pip_hbox := HBoxContainer.new()
	pip_hbox.add_theme_constant_override("separation", 3)
	for _p in range(num_slots):
		var pip := ColorRect.new()
		pip.custom_minimum_size = Vector2(10, 10)
		pip.color = C_PIP_EMPTY
		_make_circle(pip)
		pip_hbox.add_child(pip)

	return {"container": hbox, "slots": slots, "pips": pip_hbox, "index": row_index}
```

```gdscript
		if s < current_guess.size():

func _set_slot_filled(slot: Control, color: Color) -> void:
```

```gdscript
func _flip_row_reveal(row: Dictionary, history_entry: Dictionary) -> void:
	var slots: Array = row.slots
	var guess: Array = history_entry.guess
	var feedback: Dictionary = history_entry.feedback
	var stagger_ms := 80

	for s in range(slots.size()):
		var slot := slots[s] as Control
		var dot_color := PALETTE[guess[s]].color as Color

		# Phase 1: rotate 0 → 90 degrees (show dot color)
		var tw1 := create_tween()
		tw1.tween_property(slot, "rotation_degrees", 90.0, 0.1).set_trans(Tween.TRANS_SINE)
		await tw1.finished
		_set_slot_filled(slot, dot_color)

		# Phase 2: rotate 90 → 0 degrees (reveal result)
		var tw2 := create_tween()
		tw2.tween_property(slot, "rotation_degrees", 0.0, 0.1).set_trans(Tween.TRANS_SINE)
		await tw2.finished

		# Apply pip for this slot if per-dot mode (Easy)
		# For count-only modes, pips are applied all at once after flip
		if s < slots.size() - 1:
			await get_tree().create_timer(stagger_ms * 0.001).timeout

	# After all flips, update pips for count-only modes
	_update_pips(row.pips, feedback.exact, feedback.misplaced, slots.size())
```

```gdscript
const _CODE_CHARS := "ABCDEF"  # A=0, B=1, C=2, D=3, E=4, F=5

func _encode_custom_puzzle(sequence: Array[int]) -> String:
	var code := CUSTOM_PUZZLE_PREFIX
	for ci in sequence:
		code += _CODE_CHARS[ci]
	return code

func _decode_custom_puzzle(code: String) -> Array[int]:
	if not code.begins_with(CUSTOM_PUZZLE_PREFIX):
		return []
	var body := code.substr(CUSTOM_PUZZLE_PREFIX.length())
	if body.length() < 3 or body.length() > 6:
		return []
	var result: Array[int] = []
	for ch in body:
		var idx := _CODE_CHARS.find(ch.to_upper())
		if idx == -1:
			return []
		result.append(idx)
	return result
```

```gdscript
func _evaluate_guess(guess: Array, secret: Array) -> Dictionary:
	var exact := 0
	var misplaced := 0
	var per_dot: Array = []
	var secret_remaining := secret.duplicate()
	var guess_remaining := guess.duplicate()

	# First pass: exact
	for i in range(guess.size()):
		if guess[i] == secret[i]:
			exact += 1
			per_dot.append("exact")
			secret_remaining[i] = -1
			guess_remaining[i]  = -1
		else:
			per_dot.append("absent")

	# Second pass: misplaced
	for i in range(guess.size()):
		if guess_remaining[i] == -1:
			continue
		var found := secret_remaining.find(guess_remaining[i])
		if found != -1:
			misplaced += 1
			per_dot[i] = "misplaced"
			secret_remaining[found] = -1

	return {"exact": exact, "misplaced": misplaced, "per_dot": per_dot}
```
